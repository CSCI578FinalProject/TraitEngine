    public void write(char c[], int off, int len)
        throws IOException {

        if (suspended)
            return;

        conv.convert(c, off, len);
        conv.flushBuffer();
        charsWritten += len;

    }



    /**
     * Append a string to the buffer
     */
    public void write(String s, int off, int len)
        throws IOException {

        if (suspended)
            return;

        charsWritten += len;
        if (s == null)
            s = "null";
        conv.convert(s, off, len);
        conv.flushBuffer();

    }
