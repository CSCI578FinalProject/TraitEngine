    private String getMessage(final int message){
        if (org.apache.coyote.Constants.IS_SECURITY_ENABLED){
           return (String)AccessController.doPrivileged(
                new PrivilegedAction(){
                    public Object run(){
                        return HttpMessages.getMessage(message); 
                    }
                }
           );
        } else {
            return HttpMessages.getMessage(message);
        }
    }
