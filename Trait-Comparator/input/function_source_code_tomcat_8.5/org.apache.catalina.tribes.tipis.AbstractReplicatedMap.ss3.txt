    public Member getNextBackupNode() {
        Member[] members = getMapMembers();
        int node = getNextBackupIndex();

    public int getNextBackupIndex() {
        int size = mapMembers.size();
        if (mapMembers.size() == 0)return -1;
        int node = currentNode++;
        if (node >= size) {
            node = 0;
            currentNode = 0;
        }
        return node;
    }
